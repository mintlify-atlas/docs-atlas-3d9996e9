### Search Execution

The search operation uses Azure's `SearchAsync` API (/home/daytona/workspace/source/src/RagConsole/Services/AzureSearchService.cs:23):

```csharp
public async Task<IReadOnlyList<RagSource>> SearchAsync(string question, CancellationToken cancellationToken = default)
{
    var searchOptions = new SearchOptions
    {
        Size = options.AzureAISearch.Top
    };

    var response = await searchClient.SearchAsync<SearchDocument>(question, searchOptions, cancellationToken);
    var results = new List<RagSource>();

    await foreach (var item in response.Value.GetResultsAsync())
    {
        var id = GetFirstString(item.Document, "id", "chunk_id", "key") ?? Guid.NewGuid().ToString("N");
        var content = GetFirstString(item.Document, "content", "chunk", "text", "page_content") ?? string.Empty;
        var source = GetFirstString(item.Document, "source", "title", "file", "url") ?? "desconocido";
        var score = item.Score ?? 0d;

        if (!string.IsNullOrWhiteSpace(content))
        {
            results.Add(new RagSource(id, source, content, score));
        }
    }

    return results;
}
```

### Flexible Field Mapping

The service includes a helper method to handle different document schemas (/home/daytona/workspace/source/src/RagConsole/Services/AzureSearchService.cs:49):

```csharp
private static string? GetFirstString(SearchDocument document, params string[] keys)
{
    foreach (var key in keys)
    {
        if (!document.TryGetValue(key, out var value) || value is null)
        {
            continue;
        }

        var text = value.ToString();
        if (!string.IsNullOrWhiteSpace(text))
        {
            return text;
        }
    }

    return null;
}
```

<Accordion title="Why Flexible Field Mapping?">
Different indexing tools may use different field names:
- Content fields: `content`, `chunk`, `text`, `page_content`
- ID fields: `id`, `chunk_id`, `key`
- Source fields: `source`, `title`, `file`, `url`

This approach makes the system compatible with various indexing strategies without code changes.
</Accordion>

### Answer Generation

The core generation method constructs and sends chat completion requests (/home/daytona/workspace/source/src/RagConsole/Services/AzureOpenAiService.cs:19):

```csharp
public async Task<string> GenerateAnswerAsync(string question, IReadOnlyList<RagSource> sources, string systemPrompt, CancellationToken cancellationToken = default)
{
    var endpoint = options.AzureOpenAI.Endpoint.TrimEnd('/');
    var deployment = options.AzureOpenAI.ChatDeployment;
    var apiVersion = options.AzureOpenAI.ApiVersion;

    var url = $"{endpoint}/openai/deployments/{deployment}/chat/completions?api-version={apiVersion}";

    var prompt = BuildPrompt(question, sources);

    var payload = new
    {
        messages = new object[]
        {
            new { role = "system", content = systemPrompt },
            new { role = "user", content = prompt }
        },
        temperature = options.AzureOpenAI.Temperature,
        max_tokens = options.AzureOpenAI.MaxTokens
    };

    using var request = new HttpRequestMessage(HttpMethod.Post, url)
    {
        Content = new StringContent(JsonSerializer.Serialize(payload), Encoding.UTF8, "application/json")
    };

    request.Headers.Add("api-key", options.AzureOpenAI.ApiKey);

    using var response = await httpClient.SendAsync(request, cancellationToken);
    var responseBody = await response.Content.ReadAsStringAsync(cancellationToken);

    if (!response.IsSuccessStatusCode)
    {
        throw new InvalidOperationException($"Azure OpenAI devolvió {(int)response.StatusCode}: {responseBody}");
    }

    using var json = JsonDocument.Parse(responseBody);
    var answer = ExtractAnswer(json.RootElement);

    if (string.IsNullOrWhiteSpace(answer))
    {
        throw new InvalidOperationException("No se pudo extraer respuesta de Azure OpenAI.");
    }

    return answer;
}
```

### Response Extraction

The service includes robust JSON parsing to extract the answer (/home/daytona/workspace/source/src/RagConsole/Services/AzureOpenAiService.cs:94):

```csharp
private static string ExtractAnswer(JsonElement root)
{
    if (!root.TryGetProperty("choices", out var choices) || choices.ValueKind != JsonValueKind.Array || choices.GetArrayLength() == 0)
    {
        return string.Empty;
    }

    var firstChoice = choices[0];
    if (!firstChoice.TryGetProperty("message", out var message))
    {
        return string.Empty;
    }

    if (!message.TryGetProperty("content", out var content))
    {
        return string.Empty;
    }

    if (content.ValueKind == JsonValueKind.String)
    {
        return content.GetString() ?? string.Empty;
    }

    // Handle array responses (for multipart content)
    if (content.ValueKind == JsonValueKind.Array)
    {
        var parts = new List<string>();
        foreach (var item in content.EnumerateArray())
        {
            if (item.TryGetProperty("text", out var textNode) && textNode.ValueKind == JsonValueKind.String)
            {
                var part = textNode.GetString();
                if (!string.IsNullOrWhiteSpace(part))
                {
                    parts.Add(part);
                }
            }
        }

        return string.Join(Environment.NewLine, parts);
    }

    return string.Empty;
}
```

<Info>
This method handles both simple string responses and structured array responses, making it compatible with different OpenAI response formats.
</Info>

public List<string> Validate()

    if (IsMissing(AzureAISearch.Endpoint))
    {
        errors.Add("AzureAISearch.Endpoint es requerido.");

    if (IsMissing(AzureAISearch.ApiKey))

    }

    if (IsMissing(AzureAISearch.IndexName))

private static bool IsMissing(string? value)
{
    return string.IsNullOrWhiteSpace(value) || value.Contains("<") || value.Contains(">");
}

## Best Practices

1. **Use Managed Identity** in production instead of API keys when possible
2. **Set appropriate timeouts** on HttpClient for OpenAI calls
3. **Implement retry logic** for transient failures
4. **Monitor token usage** to optimize costs
5. **Cache search results** when appropriate to reduce API calls
6. **Use the latest API version** for new features and improvements

## Performance Considerations

### Latency Sources

1. **Search latency**: Typically 100-500ms depending on index size
2. **OpenAI generation**: 1-5 seconds depending on response length and model
3. **Network overhead**: Varies by region and connection

### Optimization Tips

- Deploy Azure resources in the same region
- Use connection pooling for HttpClient
- Consider streaming responses for better perceived performance
- Reduce `MaxTokens` if shorter answers are acceptable
